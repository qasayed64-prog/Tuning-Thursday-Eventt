  <script>
    const $ = s => document.querySelector(s);
    const $$ = s => Array.from(document.querySelectorAll(s));

    // Try multiple filenames until one works
    function loadImage(el, names){
      let i=0;
      const tryNext=()=>{
        if(i>=names.length){ el.alt+=" (not found)"; return; }
        const url=names[i++];
        const test=new Image();
        test.onload=()=>{ el.src=url; };
        test.onerror=tryNext;
        test.src=url+"?v="+Date.now(); // bust cache
      };
      tryNext();
    }

    const files={
      basketball:["basketball.png","Basketball.PNG","BASKETBALL.png"],
      quadrangle:["quadrangle.png","Quadrangle.PNG","QUADRANGLE.png"],
      mph:["mph.png","MPH.PNG","Mph.png"],
      cafeteria:["cafeteria.png","Cafeteria.PNG","CAFETERIA.png"]
    };

    function openModal(kind){
      if(kind==="basketball") loadImage($("#img-basketball"),files.basketball);
      if(kind==="quadrangle") loadImage($("#img-quadrangle"),files.quadrangle);
      if(kind==="mph") loadImage($("#img-mph"),files.mph);
      if(kind==="cafeteria") loadImage($("#img-cafeteria"),files.cafeteria);
      $("#"+kind+"Modal").style.display="block";
    }

    $$(".zone").forEach(z=>z.addEventListener("click",e=>{
      e.preventDefault();
      openModal(z.dataset.modal);
    }));

    $$(".modal").forEach(m=>{
      m.addEventListener("click",e=>{ if(e.target===m) m.style.display="none"; });
      m.querySelectorAll("[data-close]").forEach(b=>b.onclick=()=>m.style.display="none");
    });
  </script>
